\exercise{C-6.19}

\begin{algorithm*}
\SetKwFunction{FMain}{postfix}
\SetKwProg{Fn}{Function}{:}{}
\Fn{\FMain{Array expr}}{
    inputs $\gets $ new Stack()

    \For{elem in expr}{
	\If{elem.isOperation()}{
	    a $\gets $ inputs.pop() \\
	    b $\gets $ inputs.pop() \\
	    inputs.push(elem.apply(a, b))
	}
	\Else{
	    inputs.push(elem)
	}
    }
\Return{inputs.pop()}
}
\end{algorithm*}
