\exercise{1}

\begin{algorithm}
    \SetKwFunction{FMain}{add}
    \SetKwProg{Fn}{Function}{:}{}
    \Fn{\FMain{List L, Position j, Element e}}{
	\If{L.size() < j}{
	    M $\gets$ new List(new size) \\
	    \For{i = 1..(j-1)}{
		M[i] $\gets$ L[i]
	    }
	    M[j] $\gets $ e \\
	    \For{i = j..(L.length)}{
		M[i+1] $\gets$ L[i]
	    }
	}
	\Else{
	    $i \gets $ L.length \\
	    \While{ i > j}{
		L[i+1] $\gets $ L[i] \\
		i $\gets $ i - 1
	    }
	    L[j] $\gets $ e
	}
    }
\end{algorithm}

\begin{algorithm}[H]
    \SetKwFunction{FMain}{addLast}
    \SetKwProg{Fn}{Function}{:}{}
    \Fn{\FMain{Element e}}{
	p $\gets $ self.last() \\
	self.addAfter(p, e)
    }
\end{algorithm}

\begin{algorithm}[H] 
    \SetKwFunction{FMain}{addBefore}
    \SetKwProg{Fn}{Function}{:}{}
    \Fn{\FMain{Element e, Position p}}{
	p' $\gets $ self.before(p) \\
	self.addAfter(p', e)
    }
\end{algorithm}
